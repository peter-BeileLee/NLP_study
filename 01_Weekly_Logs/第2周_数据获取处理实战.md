```python

import requests
from bs4 import BeautifulSoup
import time

import os
import re
import numpy as np
import pandas as pd
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.cluster import KMeans


# 1. 目标 URL：古登堡计划中一页公开的英文小说
# 它的版权已经公开，属于公有领域，合法可爬
url = "https://www.gutenberg.org/files/11/11-h/11-h.htm"

# 2. 合法合规操作：设置请求头（User-Agent）
# 主动表明身份，留下你的联系邮箱（工业界规范），告诉对方我们是用于学习的良性爬虫
# headers = {
#     'User-Agent': 'NLP_Learner_Bot/1.0 (my_learing_email@example.com)'
# }

print(  "[INFO] 正在请求网页..." )
# response = requests.get(url, headers=headers)
response = requests.get(url)
# 检查网站是否允许我们访问（200 代表成功）
if response.status_code == 200:
    print(  "[INFO] 网页下载成功！正在解析文本..." )
    
    # 3. 解析网页 HTML
    soup = BeautifulSoup(response.text, 'html.parser')
    
    # 4. 提取所有的段落标签 <p>
    # 网页里的文章正文通常都包裹在 <p> </p> 标签里
    paragraphs = soup.find_all('p')
    
    # 5. 暂存数据：按照你的要求，暂时不精细整理，先全部捞出来放到一个列表中
    raw_corpus = []
    
    for i, p in enumerate(paragraphs):
        text = p.get_text().strip()
        
        # 过滤掉太短的空白行
        if len(text) > 10:
            raw_corpus.append(text)
            
            # 模拟工业界规范：每爬取几条数据就主动休息一下，保护对方服务器
            if i % 10 == 0:
                time.sleep(0.5)
                
    print(f"\n[SUCCESS] 成功捞取了 {len(raw_corpus)} 段原始文本！" )
    
    # 6. 打印前 3 段原始数据看看样子
    print("\n--- 原始数据切片展示 ---")
    for doc in raw_corpus[:3]:
        print(f"👉 {doc}\n")
        
else:
    print(f"[ERROR] 无法访问该网页，状态码：{response.status_code}")

# 定义本地保存的文件名
output_file = "alice_raw_corpus.txt"

print(f"\n[INFO] 正在将 {len(raw_corpus)} 段文本永久保存到本地 {output_file} ...")

# 使用 with open 语法安全地打开并创建本地文件
# 'w' 代表 write（写入模式），encoding='utf-8' 确保中文或特殊英文符号不会变成乱码
with open(output_file, 'w', encoding='utf-8') as f:
    for doc in raw_corpus:
        # 将每一段文本写入文件，并在末尾强制加上换行符 \n，让它们一行一段，井井有条
        f.write(doc + "\n")

print("[SUCCESS] 数据落盘成功！现在你关闭电脑，数据也不会丢失了。")


# =====================================================================
# 1. 数据加载阶段 (Data Loading)
# =====================================================================
input_file = "alice_raw_corpus.txt"

# 安全检查：确保你之前的爬虫代码确实成功生成了该文件
if not os.path.exists(input_file):
    print(f"[ERROR] 未找到本地文件 '{input_file}'，请确保你已经运行了爬虫保存代码！")
    # 如果文件不存在，这里用几句模拟数据撑一下场面，防止代码直接报错崩溃
    lines = [
        "Alice was beginning to get very tired of sitting by her sister on the bank.",
        "The Rabbit took a watch out of its waistcoat-pocket, and looked at it.",
        "Off with her head! the Queen shouted at the top of her voice.",
        "Speak roughly to your little boy, and beat him when he sneezes.",
        "A large rose-tree stood near the entrance of the garden."
    ]
    print("[WARN] 已自动启用模拟测试文本。")
else:
    print(f"[INFO] 成功检测到本地文本，正在读取 '{input_file}' ...")
    with open(input_file, 'r', encoding='utf-8') as f:
        # 逐行读取，并去掉每行末尾看不见的换行符 \n
        lines = [line.strip() for line in f.readlines()]

# 把读取出来的文本行，整整齐齐地塞进 Pandas 表格的一列中
df = pd.DataFrame(lines, columns=['raw_text'])
print(f"[SUCCESS] 成功加载本地数据！总计读取到：{len(df)} 个段落文本。\n")


# =====================================================================
# 2. 文本清洗与规范化阶段 (Text Preprocessing)
# =====================================================================
def clean_english_text(text):
    """
    专门针对英文小说文本的脱水清洗函数
    """
    # a. 统一变成小写（防止计算机认为 'Alice' 和 'alice' 是两个完全不同的词）
    text = text.lower()
    
    # b. 用正则表达式把所有不是英文字母、不是空格的杂质（如标点符号 !,.?;"等）全部替换成空格
    text = re.sub(r'[^a-zA-Z\s]', ' ', text)
    
    # c. 切碎再拼接：切碎能自动干掉多余的换行或连续空格，再用单个空格拼接回来
    words = text.split()
    
    # 返回洗得白白净净的一行字符串
    return " ".join(words)

print("[INFO] 正在启动文本规范化流水线...")
# 批量让表格里 raw_text 的每一行去调用上面这个洗菜函数，结果存到新列 cleaned_text 中
df['cleaned_text'] = df['raw_text'].apply(clean_english_text)
print("[SUCCESS] 文本规范化清洗完毕！展示前 3 段清洗对比：")
print("-" * 60)
for index, row in df.head(3).iterrows():
    print(f"原始文本: {row['raw_text'][:60]}...")
    print(f"清洗文本: {row['cleaned_text'][:60]}...")
    print("-" * 60)


# =====================================================================
# 3. 符号变矩阵：TF-IDF 特征工程 (Feature Extraction)
# =====================================================================
print("\n[INFO] 正在将人类文字转化为固定长度的数字矩阵 (TF-IDF)...")

# 初始化转换器，并开启两项工业级配置：
# stop_words='english': 自动加载内置的英文停用词表，剔除 "the", "and", "is" 等废词
# max_features=500: 只保留全书中最重要的前 500 个高频核心特征词，死死锁定特征维度 d = 500
tfidf_wizard = TfidfVectorizer(stop_words='english', max_features=500)

# 让转换器去学习整本小说的词表 (fit)，并把 cleaned_text 一键翻译成纯数字特征矩阵 (transform)
# .toarray() 是为了把稀疏矩阵转化成我们能看得懂的、整齐的 NumPy 二维数字矩阵
X = tfidf_wizard.fit_transform(df['cleaned_text']).toarray()

print(f"[SUCCESS] 特征矩阵 (X) 生成成功！")
print(f"👉 矩阵形状 (Shape) = {X.shape} -> 代表：{X.shape[0]} 行段落，每行被翻译成了 {X.shape[1]} 维的固定长度数字向量")
print(f"👉 词表字典前 10 个特征词展示：{tfidf_wizard.get_feature_names_out()[:10]}\n")


# =====================================================================
# 4. 实例应用：无监督算法聚类测试 (Machine Learning Application)
# =====================================================================
print("[INFO] 正在启动无监督机器学习（K-Means 聚类算法）...")
print("[INFO] 我们打算让计算机在没有参考答案的情况下，自动把这些段落划分为 3 个大类...")

# 初始化一个 K-Means 聚类器，指定分成 3 个堆 (Clusters)
kmeans_model = KMeans(n_clusters=3, random_state=42, n_init='auto')

# 让算法大口吞下我们刚才做好的数字特征矩阵 X，开始在 500 维的几何空间里算距离、搞聚类 (Fit)
kmeans_model.fit(X)

# 算法算完后，会给每一行文本打上一个它自认为分类合理的标签 (0, 1, 或 2)
df['cluster_label'] = kmeans_model.labels_

print("[SUCCESS] 聚类学习完成！以下是计算机自动分出的 3 个类别样本抽样展示：")
print("=" * 60)

# 循环把 3 个类别的结果各抽 2 句话打印出来看看
for cluster_id in range(3):
    print(f"🤖 计算机自动划分出的【类别 {cluster_id}】文本抽样：")
    sample_texts = df[df['cluster_label'] == cluster_id]['raw_text'].head(2).values
    for text in sample_texts:
        print(f"  - {text[:80]}...")
    print("-" * 60)

```
