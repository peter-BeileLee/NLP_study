## 🐼 Pandas 速查（数据清洗与表格处理）

```python
import pandas as pd

# --- 数据导入 ---
df = pd.read_csv('file.csv')          # 读取 CSV 文件到 DataFrame
df = pd.read_json('file.json')        # 读取 JSON 文件到 DataFrame

# --- 数据看盘 ---
df.head(5)                            # 查看前 5 行数据
df.shape                              # 返回(行数, 列数)元组
df.info()                             # 查看各列数据类型与缺失值基本信息
df.describe()                         # 查看数值列的统计指标(均值、最值等)

# --- 数据筛选 (手术刀) ---
df[df['age'] > 18]                    # 单条件筛选：行过滤
df[(df['age'] > 18) & (df['score'] > 90)] # 多条件筛选：&代表且，|代表或，条件必须加括号
df[['name', 'age']]                   # 列筛选：只保留指定的几列

# --- 统计与分析 ---
df['label'].value_counts()            # 类别分布统计：统计某列各类别出现次数
df['label'].value_counts(normalize=True) # 类别比例统计：看各类别百分比占比
df['text'].str.len()                  # 文本长度统计：计算字符串列每个元素的字数
df.groupby('city')['spend'].mean()    # 分组统计：按city分组，计算spend的平均值
df.groupby('city')['spend'].agg(['mean', 'sum']) # 多维分组：同时计算均值和总和

# --- 数据清洗 ---
df.dropna(subset=['col'])             # 删除指定列含有空值(NaN)的行
df.fillna(0)                          # 将表中所有空值(NaN)填充为 0

```

## 🔢 NumPy 速查（底层矩阵与数值计算）

```python
import numpy as np

# --- 创建数组/矩阵 ---
arr = np.array([1, 2, 3])             # 从 Python 列表创建一维数组
matrix = np.array([[1,2], [3,4]])     # 从嵌套列表创建二维矩阵
zeros = np.zeros((3, 4))              # 创建 3行4列 全为 0 的矩阵
ones = np.ones((2, 3))                # 创建 2行3列 全为 1 的矩阵
arange = np.arange(0, 10, 2)          # 创建等差数组 [0, 2, 4, 6, 8]

# --- 矩阵形状与变换 ---
arr.shape                             # 返回数组/矩阵的维度形状
matrix.reshape(4, 1)                  # 改变矩阵形状为 4行1列 (总元素量须保持一致)
matrix.T                              # 矩阵转置 (行列对调)

# --- 数学与统计计算 ---
np.mean(arr)                          # 计算数组所有元素的平均值
np.sum(matrix, axis=0)                # 矩阵压扁求和：axis=0 按列求和，axis=1 按行求和
np.max(arr)                           # 寻找数组中的最大值
np.argmax(arr)                        # 寻找最大值所在的索引位置(位置下标)
```
